or_mod <- gam(days_eff_adj ~ 
                    s(month, bs = "cc", k = 10) + #adjusted from 12 - make sure is ok
                    s(year, bs = "re"),
                  family = tw(link = "log"),
                  data = or_zc,
                  method = "fREML")


or_grid <- expand.grid(
  month = seq(0, 13, by = 0.1),   # extend beyond 1-12
  year  = 0)

pred <- predict(or_mod,
                newdata = or_grid,
                exclude = "s(year)",
                newdata.guaranteed = TRUE,
                type = "link",
                se.fit = TRUE)

or_grid <- or_grid %>%
  mutate(
    fit = exp(pred$fit),
    lwr = exp(pred$fit - 1.96 * pred$se.fit),
    upr = exp(pred$fit + 1.96 * pred$se.fit)
  ) %>% 
  mutate(across(fit:upr, \(x) x / pracma::trapz(month, fit), .names = "{.col}_norm"))

or_peaks <-  find_peaks(or_grid)
